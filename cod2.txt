$Simulacion de acceso 
gantz edad, permiso; 
kette mensaje;

edad = 17; 
permiso = 1;

wen ((edad >= 18) und (permiso == 1)) { 
    
    mensaje = "acceso permitido";
} sonwen ((edad >= 18) und (permiso == 0)) { 
    
    mensaje = "falta permiso";
} sonwen ((edad < 18) und (permiso == 1)) { 
    
    mensaje = "menor con permiso especial";

    mensaje = "acceso denegado"; 
}
